@router.post("/resources/", response_model=ResourceSchema)
def create_resource(
    request: ResourceCreate, 
    db: Session = Depends(get_db), 
    current_user: User = Depends(get_current_active_user)
):
    # 1. Identity Verification
    if not current_user:
        raise HTTPException(status_code=401, detail="Unauthorized")

    # 2. Persist Intent (Transactional)
    new_resource = Resource(
        user_id=current_user.id,
        name=request.name,
        provider=request.provider,
        type=request.type,
        status="pending",
        configuration=request.variables
    )
    db.add(new_resource)
    db.commit()
    db.refresh(new_resource)

    # 3. Queue Asynchronous Execution
    provision_resource_task.delay(
        resource_id=new_resource.id,
        user_id=current_user.id,
        provider=request.provider,
        module=request.type,
        vars=request.variables
    )
    
    return new_resource
\end{lstlisting}

\section{Security: The Encryption Service}
\begin{lstlisting}[language=Python, caption=AES-128 Fernet Implementation]
from cryptography.fernet import Fernet
import os

def get_vault_engine():
    key = os.getenv("SECRET_KEY")
    if not key:
        raise ValueError("SECRET_KEY not found in environment")
    return Fernet(key)

def encrypt_credential(plain_text: str) -> str:
    f = get_vault_engine()
    return f.encrypt(plain_text.encode()).decode()

def decrypt_credential(cipher_text: str) -> str:
    f = get_vault_engine()
    return f.decrypt(cipher_text.encode()).decode()
\end{lstlisting}

\section{Backend: The Background Provisioning Worker}
\begin{lstlisting}[language=Python, caption=Celery Worker Provisioning Flow]
@celery.task(bind=True, max_retries=3, default_retry_delay=30)
def provision_resource_task(self, resource_id: int, user_id: int, provider: str, module: str, vars: dict):
    db = SessionLocal()
    resource = db.query(Resource).filter(Resource.id == resource_id).first()

    try:
        resource.status = "running"
        db.commit()

        credential = load_credential_for_user(db, user_id, provider)
        secrets = decrypt_provider_bundle(credential)
        workspace = prepare_workspace(resource_id, provider, module, vars)

        write_provider_files(workspace, provider, module, vars)
        tf_env = build_terraform_env(secrets)

        init_result = run_command(["terraform", "init"], cwd=workspace, env=tf_env)
        if init_result.returncode != 0:
            raise RuntimeError(init_result.stderr)

        apply_result = run_command(
            ["terraform", "apply", "-auto-approve"],
            cwd=workspace,
            env=tf_env,
        )

        resource.execution_log = apply_result.stdout + "\n" + apply_result.stderr
        resource.status = "active" if apply_result.returncode == 0 else "failed"
        db.commit()

        if resource.status == "failed":
            analyze_with_ai_copilot.delay(resource.id)
    except RetryableCloudError as exc:
        resource.status = "retry_scheduled"
        db.commit()
        raise self.retry(exc=exc)
    except Exception as exc:
        resource.status = "failed"
        resource.execution_log = str(exc)
        db.commit()
        raise
    finally:
        db.close()
\end{lstlisting}

\section{Backend: Inventory Synchronization Task}
\begin{lstlisting}[language=Python, caption=Periodic Multi-Cloud Sync Task]
@celery.task
def sync_all_resources():
    db = SessionLocal()
    credentials = db.query(CloudCredential).all()

    for credential in credentials:
        provider = credential.provider_type.lower()
        decrypted = decrypt_provider_bundle(credential)

        if provider == "aws":
            resources = fetch_aws_inventory(decrypted)
        elif provider == "azure":
            resources = fetch_azure_inventory(decrypted)
        elif provider == "gcp":
            resources = fetch_gcp_inventory(decrypted)
        else:
            continue

        for payload in resources:
            canonical = normalize_resource(provider, payload)
            upsert_resource_record(db, credential.user_id, canonical)

    db.commit()
    db.close()
\end{lstlisting}

\section{Frontend: Resource Status Polling Hook}
\begin{lstlisting}[language=Python, caption=React Hook for Task Polling]
export const useTaskPolling = (taskId?: string) => {
  const [status, setStatus] = useState("idle");
  const [log, setLog] = useState("");

  useEffect(() => {
    if (!taskId) return;

    const timer = setInterval(async () => {
      const response = await axios.get(`/api/tasks/${taskId}`);
      setStatus(response.data.status);
      setLog(response.data.execution_log ?? "");

      if (["completed", "failed", "active"].includes(response.data.status)) {
        clearInterval(timer);
      }
    }, 3000);

    return () => clearInterval(timer);
  }, [taskId]);

  return { status, log };
};
\end{lstlisting}
