% Appendix C: Full HCL Templates
\chapter{FULL HCL TEMPLATES}

This appendix provides the complete Terraform HCL (HashiCorp Configuration Language) templates used by the Nebula engine for resource provisioning across AWS, Azure, and GCP.

\section{AWS VPC and Subnet Configuration}
\begin{lstlisting}[language=XML, caption=AWS Infrastructure (vpc.tf)]
resource "aws_vpc" "nebula_main" {
  cidr_block           = "10.0.0.0/16"
  enable_dns_support   = true
  enable_dns_hostnames = true
  
  tags = {
    Name = "nebula-vpc"
  }
}

resource "aws_subnet" "nebula_public" {
  vpc_id                  = aws_vpc.nebula_main.id
  cidr_block              = "10.0.1.0/24"
  map_public_ip_on_launch = true
  
  tags = {
    Name = "nebula-public-subnet"
  }
}
\end{lstlisting}

\section{Azure Resource Group and Network Configuration}
\begin{lstlisting}[language=XML, caption=Azure Infrastructure (network.tf)]
resource "azurerm_resource_group" "nebula_rg" {
  name     = "nebula-resources"
  location = "East US"
}

resource "azurerm_virtual_network" "nebula_vnet" {
  name                = "nebula-vnet"
  address_space       = ["10.0.0.0/16"]
  location            = azurerm_resource_group.nebula_rg.location
  resource_group_name = azurerm_resource_group.nebula_rg.name
}
\end{lstlisting}

% Appendix E: Database DDL Statements
\chapter{DATABASE DDL STATEMENTS}

This appendix provides the SQL (PostgreSQL) DDL (Data Definition Language) statements used to initialize the platform's persistence layer.

\section{Table: Users}
\begin{lstlisting}[language=SQL]
CREATE TABLE users (
    id SERIAL PRIMARY KEY,
    username VARCHAR(50) UNIQUE NOT NULL,
    hashed_password TEXT NOT NULL,
    email VARCHAR(100) UNIQUE NOT NULL,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);
\end{lstlisting}

\section{Table: Cloud Credentials (Encrypted)}
\begin{lstlisting}[language=SQL]
CREATE TABLE cloud_credentials (
    id SERIAL PRIMARY KEY,
    user_id INTEGER REFERENCES users(id),
    provider_type VARCHAR(20) NOT NULL,
    access_key_enc TEXT NOT NULL,
    secret_key_enc TEXT NOT NULL,
    tenant_id_enc TEXT,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);
\end{lstlisting}

\section{Table: Resources}
\begin{lstlisting}[language=SQL]
CREATE TABLE resources (
    id SERIAL PRIMARY KEY,
    user_id INTEGER REFERENCES users(id),
    provider VARCHAR(20) NOT NULL,
    resource_type VARCHAR(50) NOT NULL,
    status VARCHAR(20) DEFAULT 'pending',
    provider_id VARCHAR(100),
    metadata JSONB,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);
\end{lstlisting}
